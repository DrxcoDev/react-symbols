```jsx
import React from 'react';
import { Angular, Astro, Document } from '@react-symbols/icons';

const MyComponent = () => {
  return (

      <Angular width={128} height={128} />
      <Astro width={128} height={128} />
      <Document width={128} height={128} />
    </>
  );
};

export default MyComponent;
```
